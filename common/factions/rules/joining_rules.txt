joining_rule_non_fascist = {
	type = joining_rules
	visible = {
		NOT = {
			has_government = purificationism
			has_government = technocratic_despotism
			has_government = autocracy
		}
	}
	ai_will_do = {
		base = 0
		modifier = {
			add = 1
			NOT = {
				OR = {
					has_government = purificationism
					has_government = technocratic_despotism
					has_government = autocracy
				}
			}
		}
		modifier = {
			add = 1
			OR = {
				has_government = katzerism
				has_government = autonomous_socialism
			}
		}
	}
	trigger = {
		NOT = {
			OR = {
				has_government = purificationism
				has_government = technocratic_despotism
				has_government = autocracy
			}
		}
		IF = {
			limit = {
				NOT = {
					has_rule = can_join_opposite_factions
				}
			}
			has_government = FROM
		}
	}
}

joining_rule_non_communist = {
	type = joining_rules
	visible = {
		NOT = {
			OR = {
				has_government = neosocialism
				has_government = katzerism
				has_government = autonomous_socialism
			}
		}
	}
	ai_will_do = {
		base = 0
		modifier = {
			add = 1
			NOT = {
				OR = {
					has_government = neosocialism
					has_government = katzerism
					has_government = autonomous_socialism
				}
			}
		}
		modifier = {
			add = 1
			OR = {
				has_government = legalism_aristocratic_democracy_subtype
				has_government = legalism_facade_democracy_subtype
				has_government = legalism_nepotist_respublicanism_subtype
				has_government = autocracy
			}
		}
	}
	available = {
		IF = {
			limit = {
				has_faction_template = faction_template_freedom_accord
			}
			NOT = {
				any_enemy_country = {
					is_major = yes
					OR = {
						has_government = technocratic_despotism
						has_government = purificationism
					}
				}
			}
		}
	}
	trigger = {
		NOT = {
			OR = {
				has_government = neosocialism
				has_government = katzerism
				has_government = autonomous_socialism
			}
		}
		IF = {
			limit = {
				NOT = {
					has_rule = can_join_opposite_factions
				}
			}
			has_government = FROM
		}
	}
}

		}
	}
}
