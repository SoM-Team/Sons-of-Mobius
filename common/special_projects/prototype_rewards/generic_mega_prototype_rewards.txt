sp_mega_major_political_interference_generic_reward = {
	fire_only_once = yes
	threshold = {
		min = 15
		max = 85
	}
	weight = {
		base = 1
		modifier = {
			factor = 0.25
			FROM = {
				NOT = {
					has_government = purificationism
					has_government = technocratic_despotism
					has_government = autocracy
					has_government = neosocialism
				}
			}
		}
	}
	option = {
		token = sp_mega_major_political_interference_generic_reward_option_keep
		iteration_output = {
			country_effects = {
				add_political_power = -50
				FROM = {
					add_project_progress_ratio = constant:sp_progress.loss.medium
				}
			}
			scientist_effects = {
				add_scientist_xp = {
					experience = constant:sp_scientist_xp_gain.medium
					specialization = specialization_mega_projects
				}
			}
		}
	}
	option = {
		token = sp_mega_major_political_interference_generic_reward_option_fire
		iteration_output = {
			country_effects = {
				add_political_power = 50
				FROM = {
					add_project_progress_ratio = constant:sp_progress.loss.low
				}
			}
			scientist_effects = {
				remove_scientist_role = yes
			}
		}
	}
}

#Low Scientist XP gain
sp_mega_generic_reward_scientist_xp_1 = {

	fire_only_once = yes

	threshold = {
		min = 0
		max = 100
	}

	# ROOT is the project
	# FROM is the Country
	# var:facility_state is the State where the facility is (ensured to be set in this case)
	# var:facility_province_id is the province ID where the facility is (ensured to be set in this case)
	# var:scientist is the Scientist (ensured to be set in this case)
	weight = {
		base = @base_reward_weight
		modifier = {
			factor = @reward_weight_factor_theme
			has_project_flag = sp_mega_generic_reward_scientist_xp_flag
		}
		modifier = {
			factor = 0
			var:scientist = {
				has_scientist_level = {
					level > 4
					specialization = specialization_mega_projects
				}
			}
		}
	}
	
	option = {
		token = sp_mega_generic_reward_option_scientist_xp
		
		iteration_output = {
			scientist_effects = {
				if = {
					limit = {
						
					}
					add_scientist_xp = {
						experience = constant:sp_scientist_xp_gain.low
						specialization = specialization_mega_projects
					}
				}
				FROM = { set_project_flag = sp_mega_generic_reward_scientist_xp_flag }
			}
		}
	}
}
